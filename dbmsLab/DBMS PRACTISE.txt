(
	COURSE_CODE INT,
	CREDIT NUMBER,
	INSTRUCTOR INT,
	CONSTRAINT courses_table_pk PRIMARY KEY (COURSE_CODE),
	CONSTRAINT courses_table_fk FOREIGN KEY (INSTRUCTOR) REFERENCES INSTRUCTORS(INST_ID)
